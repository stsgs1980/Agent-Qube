   834→      ctx.beginPath(); ctx.arc(epx, epy, 6, 0, Math.PI*2);
   835→      ctx.fillStyle = 'rgba(6,182,212,' + (0.15 * alpha) + ')'; ctx.fill();
   836→      // particle core

   838→      ctx.fillStyle = 'rgba(6,182,212,' + (0.8 * alpha) + ')'; ctx.fill();
   839→    });
   840→    ctx.restore();
   841→
   842→    // -- LAYER 3: Nodes (foreground, fastest parallax) --

   844→    ctx.save();

   846→    nodes.forEach(function(n) {

   854→      // ring pulse (for hovered node)
   855→      if (hoveredNode === n) {
   856→        var pulse = 0.5 + 0.5 * Math.sin(time * 4);
   857→        ctx.beginPath(); ctx.arc(n.x, n.y, n.r + 8 + pulse*4, 0, Math.PI*2);
   858→        ctx.strokeStyle = hexRGBA(roles[n.role], 0.2 * pulse);
   859→        ctx.lineWidth = 1.5;
   860→        ctx.stroke();
   861→      }
   862→      // circle bg
   863→      ctx.beginPath(); ctx.arc(n.x, n.y, n.r, 0, Math.PI*2);
   864→      ctx.fillStyle = 'rgba(10,10,10,' + alpha + ')';
   865→      ctx.fill();
   866→      ctx.strokeStyle = hexRGBA(roles[n.role], alpha * 0.8);
   867→      ctx.lineWidth = 2;
   868→      ctx.stroke();
   869→      // role label inside

   876→      ctx.fillStyle = 'rgba(228,228,231,' + alpha + ')';
